use com_rs::ComPtr;
use directx_sys::dxgi::*;
use winapi::{LARGE_INTEGER, LUID, REFGUID, SIZE_T, UINT};

use super::Output;

#[derive(Debug)]
pub struct Adapter {
    raw: ComPtr<IDXGIAdapter>,
    desc: AdapterDesc
}

pub struct OutputIter {
    adapter: ComPtr<IDXGIAdapter>,
    index: UINT
}

impl Iterator for OutputIter {
    type Item = Output;
    fn next(&mut self) -> Option<Output> {
        let mut output: ComPtr<IDXGIOutput> = ComPtr::new();
        match unsafe {
            self.adapter.enum_outputs(self.index, output.as_mut())
        } {
            0 => {
                self.index += 1;
                unsafe { Some(Output::from_raw(output)) }
            },
            _ => None
        }
    }
}

impl Adapter {
    pub unsafe fn from_raw(raw: ComPtr<IDXGIAdapter>) -> Adapter {
        assert!(!raw.is_null());

        let mut desc = AdapterDesc::default();
        raw.get_desc(&mut desc);

        Adapter {
            raw: raw,
            desc: desc
        }
    }

    /// Returns an iterator over all available outputs.
    pub fn outputs(&self) -> OutputIter {
        OutputIter {
            adapter: self.raw.clone(),
            index: 0
        }
    }

    /// Check to see if an interface is supported on this adapter.
    ///
    /// If the interface is supported, `Ok` is returned, containing the
    /// supported version. If the interface is not supported, `Err` is returned.
    pub fn check_interface_support(&self, iid: REFGUID)
                                   -> Option<LARGE_INTEGER> {
        let mut version = 0i64;
        match unsafe { self.raw.check_interface_support(iid, &mut version) } {
            0 => Some(version),
            _ => None
        }
    }


    pub fn description(&self) -> String {
        let len = self.desc.description.iter().position(|&b| b == 0)
                                              .unwrap_or(128);
        String::from_utf16(&self.desc.description[..len]).unwrap()
    }

    pub fn vendor_id(&self) -> UINT {
        self.desc.vendor_id
    }

    pub fn device_id(&self) -> UINT {
        self.desc.device_id
    }

    pub fn subsystem_id(&self) -> UINT {
        self.desc.subsys_id
    }

    pub fn revision(&self) -> UINT {
        self.desc.revision
    }

    pub fn dedicated_video_memory(&self) -> SIZE_T {
        self.desc.dedicated_video_memory
    }

    pub fn dedicated_system_memory(&self) -> SIZE_T {
        self.desc.dedicated_system_memory
    }

    pub fn shared_system_memory(&self) -> SIZE_T {
        self.desc.shared_system_memory
    }

    pub fn adapter_luid(&self) -> LUID {
        self.desc.adapter_luid
    }
}
